017package org.microbean.configuration.cdi.annotation;
018
019import java.io.Serializable; // for javadoc only
020
021import java.lang.annotation.Documented;
022import java.lang.annotation.ElementType;
023import java.lang.annotation.Retention;
024import java.lang.annotation.RetentionPolicy;
025import java.lang.annotation.Target;
026
027import java.util.ArrayList;
028import java.util.Collection;
029
030import javax.enterprise.util.AnnotationLiteral;
031import javax.enterprise.util.Nonbinding;
032
033import javax.inject.Qualifier;
034
035/**
036 * A {@link Qualifier} annotation indicating that the annotated
037 * element is related to a particular configuration value in some way.
038 *
039 * <p>In the case of method parameters and instance fields, this
040 * usually means that the annotated element would like to receive a
041 * particular configuration value from the microBean Configuration
042 * framework.</p>
043 *
044 * @author <a href="https://about.me/lairdnelson"
045 * target="_parent">Laird Nelson</a>
046 *
047 * @see ConfigurationCoordinate
048 *
049 * @see Configuration
050 */
051@Documented
052@Qualifier
053@Retention(RetentionPolicy.RUNTIME)
054@Target({ ElementType.FIELD, ElementType.METHOD, ElementType.PARAMETER, ElementType.TYPE })
055public @interface ConfigurationValue {
056
057
058  /*
059   * Static fields.
060   */
061
062
063  /**
064   * A {@link String} that stands in for {@code null}&mdash;annotation
065   * elements cannot be set to {@code null} so if either the {@link
066   * #value()} or {@link #defaultValue()} elements in this class
067   * return the value of this field it is to be interpreted as if it
068   * were actually {@code null}.
069   *
070   * @see #value()
071   *
072   * @see #defaultValue()
073   */
074  public static final String NULL = "\0";
075
076
077  /*
078   * Elements.
079   */
080
081
082  /**
083   * The names of the configuration value in question.
084   *
085   * <p>Each name in the returned array represents a name of a
086   * configuration property whose value will be sought in order.</p>
087   *
088   * <p>If this element is not specified, or if it is an empty array,
089   * the sole effective configuration value name will be the name of
090   * the annotated element upon which this {@link ConfigurationValue}
091   * appears.</p>
092   *
093   * <p>This element will never return {@code null}.</p>
094   *
095   * @return the name of the configuration value in question; never
096   * {@code null} or {@link ConfigurationValue#NULL NULL}
097   *
098   * @see #NULL
099   */
100  @Nonbinding
101  String[] value() default {};
102
103  /**
104   * The {@link String} representation of the default value to use in
105   * case no suitable configuration value may be found.
106   *
107   * <p>This element may return {@link #NULL NULL}.</p>
108   *
109   * @return the {@link String} representation of the default value to
110   * use in case no suitable configuration value may be found, or
111   * {@link #NULL NULL}
112   *
113   * @see #NULL
114   */
115  @Nonbinding
116  String defaultValue() default NULL;
117
118
119  /*
120   * Inner and nested classes.
121   */
122  
123
124  /**
125   * An {@link AnnotationLiteral} representing a runtime instance of
126   * the {@link ConfigurationValue} annotation.
127   *
128   * @author <a href="https://about.me/lairdnelson"
129   * target="_parent">Laird Nelson</a>
130   */
131  public static final class Literal extends AnnotationLiteral<ConfigurationValue> implements ConfigurationValue {
132
133
134    /*
135     * Static fields.
136     */
137    
138
139    /**
140     * A {@link ConfigurationValue} whose {@link
141     * ConfigurationValue#value()} returns the empty string.
142     *
143     * @deprecated This {@link ConfigurationValue} is essentially
144     * useless and is slated for removal.
145     */
146    @Deprecated
147    public static final ConfigurationValue INSTANCE = of("");
148
149    /**
150     * The version of this class for {@linkplain Serializable
151     * serialization purposes}.
152     *
153     * @see Serializable
154     */
155    private static final long serialVersionUID = 1L;
156
157
158    /*
159     * Instance fields.
160     */
161
162
163    private final String[] names;
164
165    /**
166     * The {@link String} representation of the default value to use
167     * for the configuration value in question if no suitable
168     * configuration value could be found.
169     *
170     * <p>This field will never be {@code null} but may be {@link
171     * ConfigurationValue#NULL NULL}.</p>
172     *
173     * @see #defaultValue()
174     *
175     * @see ConfigurationValue#NULL
176     */
177    private final String defaultValue;
178
179
180    /*
181     * Constructors.
182     */
183
184
185    /**
186     * Creates a new {@link ConfigurationValue.Literal}.
187     *
188     * @param name the name of the configuration value; may be {@code
189     * null} or {@link ConfigurationValue#NULL NULL} in which case an
190     * empty array will be used instead
191     *
192     * @param defaultValue the {@link String} representation of the
193     * default value to use when a suitable configuration value cannot
194     * be found; may be {@code null} in which case {@link
195     * ConfigurationValue#NULL NULL} will be used instead
196     *
197     * @see #value()
198     *
199     * @see #defaultValue()
200     *
201     * @see #Literal(String[], String)
202     */
203    private Literal(final String name, final String defaultValue) {
204      this(name == null ? new String[0] : NULL.equals(name) ? new String[0] : new String[] { name }, defaultValue);
205    }
206
207    /**
208     * Creates a new {@link ConfigurationValue.Literal}.
209     *
210     * @param names the names of the configuration value; may be
211     * {@code null} in which case an empty array will be used instead
212     *
213     * @param defaultValue the {@link String} representation of the
214     * default value to use when a suitable configuration value cannot
215     * be found; may be {@code null} in which case {@link
216     * ConfigurationValue#NULL NULL} will be used instead
217     *
218     * @see #value()
219     *
220     * @see #defaultValue()
221     */
222    private Literal(final String[] names, final String defaultValue) {
223      super();
224      if (names == null || names.length <= 0) {
225        this.names = new String[0];
226      } else {
227        final Collection<String> nameCollection = new ArrayList<>();
228        for (final String name : names) {
229          if (name != null && !name.isEmpty() && !name.equals(NULL)) {
230            nameCollection.add(name);
231          }
232        }
233        this.names = nameCollection.toArray(new String[nameCollection.size()]);
234      }
235      this.defaultValue = defaultValue == null ? NULL : defaultValue;
236    }
237
238
239    /*
240     * Instance methods.
241     */
242
243    
244    /**
245     * Returns the {@link String} representation of the default value to use
246     * for the configuration value in question if no suitable
247     * configuration value could be found.
248     *
249     * <p>This method will never return {@code null} but may return
250     * {@link ConfigurationValue#NULL NULL}.</p>
251     *
252     * @return the {@link String} representation of the default value to use
253     * for the configuration value in question if no suitable
254     * configuration value could be found; never {@code null}
255     * 
256     * @see ConfigurationValue#defaultValue()
257     *
258     * @see ConfigurationValue#NULL
259     */
260    @Override
261    public final String defaultValue() {
262      return this.defaultValue;
263    }
264
265    /**
266     * Returns the names of the configuration value in question.
267     *
268     * <p>This field will never return {@code null}.</p>
269     *
270     * @return the names of the configuration value in question; never
271     * {@code null}
272     *
273     * @see ConfigurationValue#value()
274     */
275    @Override
276    public final String[] value() {
277      final String[] returnValue;
278      if (this.names.length <= 0) {
279        returnValue = this.names;
280      } else {
281        final String[] array = new String[this.names.length];
282        System.arraycopy(this.names, 0, array, 0, this.names.length);
283        returnValue = array;
284      }
285      return returnValue;
286    }
287
288
289    /*
290     * Static methods.
291     */
292
293
294    /**
295     * Returns a new {@link Literal} representing a {@link
296     * ConfigurationValue} instance whose {@link
297     * ConfigurationValue#value()} method will return the supplied
298     * {@code name}.
299     *
300     * <p>This method never returns {@code null}.</p>
301     *
302     * @param name the name of the configuration value in question;
303     * may be {@code null} or {@link ConfigurationValue#NULL NULL} in
304     * which case {@code ""} will be used instead
305     *
306     * @return a non-{@code null} {@link Literal}
307     *
308     * @see ConfigurationValue#value()
309     */
310    public static final Literal of(final String name) {
311      return new Literal(name, NULL);
312    }
313
314    /**
315     * Returns a new {@link Literal} representing a {@link
316     * ConfigurationValue} instance whose {@link
317     * ConfigurationValue#value()} method will return the supplied
318     * {@code names}.
319     *
320     * <p>This method never returns {@code null}.</p>
321     *
322     * @param names the names of the configuration value in question;
323     * may be {@code null} in which case an empty array will be used
324     * instead
325     *
326     * @return a non-{@code null} {@link Literal}
327     *
328     * @see ConfigurationValue#value()
329     */
330    public static final Literal of(final String[] names) {
331      return new Literal(names, NULL);
332    }
333    
334    /**
335     * Returns a new {@link Literal} representing a {@link
336     * ConfigurationValue} instance whose {@link
337     * ConfigurationValue#value()} method will return an array
338     * containing only the supplied {@code name} and whose {@link
339     * ConfigurationValue#defaultValue()} method will return the
340     * supplied {@code defaultValue}.
341     *
342     * <p>This method never returns {@code null}.</p>
343     *
344     * @param name the name of the configuration value in question;
345     * may be {@code null} or {@link ConfigurationValue#NULL NULL} in
346     * which case {@code ""} will be used instead
347     *
348     * @param defaultValue the default value for the configuration
349     * value in question; may be {@code null} in which case {@link
350     * ConfigurationValue#NULL NULL} will be used instead
351     *
352     * @return a non-{@code null} {@link Literal}
353     *
354     * @see ConfigurationValue#value()
355     *
356     * @see ConfigurationValue#defaultValue()
357     */
358    public static final Literal of(final String name, final String defaultValue) {
359      return new Literal(name, defaultValue);
360    }
361
362    /**
363     * Returns a new {@link Literal} representing a {@link
364     * ConfigurationValue} instance whose {@link
365     * ConfigurationValue#value()} method will return an array
366     * containing only the supplied {@code names} and whose {@link
367     * ConfigurationValue#defaultValue()} method will return the
368     * supplied {@code defaultValue}.
369     *
370     * <p>This method never returns {@code null}.</p>
371     *
372     * @param names the names of the configuration value in question;
373     * may be {@code null} or in which case an empty array will be
374     * used instead
375     *
376     * @param defaultValue the default value for the configuration
377     * value in question; may be {@code null} in which case {@link
378     * ConfigurationValue#NULL NULL} will be used instead
379     *
380     * @return a non-{@code null} {@link Literal}
381     *
382     * @see ConfigurationValue#value()
383     *
384     * @see ConfigurationValue#defaultValue()
385     */
386    public static final Literal of(final String[] names, final String defaultValue) {
387      return new Literal(names, defaultValue);
388    }
389    
390  }
