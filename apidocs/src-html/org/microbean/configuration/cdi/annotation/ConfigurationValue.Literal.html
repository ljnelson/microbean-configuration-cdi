017package org.microbean.configuration.cdi.annotation;
018
019import java.io.Serializable; // for javadoc only
020
021import java.lang.annotation.Documented;
022import java.lang.annotation.ElementType;
023import java.lang.annotation.Retention;
024import java.lang.annotation.RetentionPolicy;
025import java.lang.annotation.Target;
026
027import javax.enterprise.util.AnnotationLiteral;
028import javax.enterprise.util.Nonbinding;
029
030import javax.inject.Qualifier;
031
032/**
033 * A {@link Qualifier} annotation indicating that the annotated
034 * element is related to a particular configuration value in some way.
035 *
036 * <p>In the case of method parameters and instance fields, this
037 * usually means that the annotated element would like to receive a
038 * particular configuration value from the MicroBean Configurations
039 * framework.</p>
040 *
041 * @author <a href="https://about.me/lairdnelson"
042 * target="_parent">Laird Nelson</a>
043 *
044 * @see ConfigurationCoordinate
045 *
046 * @see Configuration
047 */
048@Documented
049@Qualifier
050@Retention(RetentionPolicy.RUNTIME)
051@Target({ ElementType.FIELD, ElementType.METHOD, ElementType.PARAMETER, ElementType.TYPE })
052public @interface ConfigurationValue {
053
054
055  /*
056   * Static fields.
057   */
058
059
060  /**
061   * A {@link String} that stands in for {@code null}&mdash;annotation
062   * elements cannot be set to {@code null} so if either the {@link
063   * #value()} or {@link #defaultValue()} elements in this class
064   * return the value of this field it is to be interpreted as if it
065   * were actually {@code null}.
066   *
067   * @see #value()
068   *
069   * @see #defaultValue()
070   */
071  public static final String NULL = "\0";
072
073
074  /*
075   * Elements.
076   */
077
078
079  /**
080   * The name of the configuration value in question.
081   *
082   * <p>If this element is not specified, the effective configuration
083   * value name will be the name of the annotated element upon which this
084   * {@link ConfigurationValue} appears.</p>
085   *
086   * <p>This element will never return {@link #NULL NULL}.</p>
087   *
088   * @return the name of the configuration value in question; never
089   * {@code null} or {@link ConfigurationValue#NULL NULL}
090   *
091   * @see #NULL
092   */
093  @Nonbinding
094  String value() default "";
095
096  /**
097   * The {@link String} representation of the default value to use in
098   * case no suitable configuration value may be found.
099   *
100   * <p>This element may return {@link #NULL NULL}.</p>
101   *
102   * @return the {@link String} representation of the default value to
103   * use in case no suitable configuration value may be found, or
104   * {@link #NULL NULL}
105   *
106   * @see #NULL
107   */
108  @Nonbinding
109  String defaultValue() default NULL;
110
111
112  /*
113   * Inner and nested classes.
114   */
115  
116
117  /**
118   * An {@link AnnotationLiteral} representing a runtime instance of
119   * the {@link ConfigurationValue} annotation.
120   *
121   * @author <a href="https://about.me/lairdnelson"
122   * target="_parent">Laird Nelson</a>
123   */
124  public static final class Literal extends AnnotationLiteral<ConfigurationValue> implements ConfigurationValue {
125
126
127    /*
128     * Static fields.
129     */
130    
131
132    /**
133     * A {@link ConfigurationValue} whose {@link
134     * ConfigurationValue#value()} returns the empty string.
135     *
136     * @deprecated This {@link ConfigurationValue} is essentially
137     * useless and is slated for removal.
138     */
139    @Deprecated
140    public static final ConfigurationValue INSTANCE = of("");
141
142    /**
143     * The version of this class for {@linkplain Serializable
144     * serialization purposes}.
145     *
146     * @see Serializable
147     */
148    private static final long serialVersionUID = 1L;
149
150
151    /*
152     * Instance fields.
153     */
154
155
156    /**
157     * The name of the configuration value in question.
158     *
159     * <p>This field will never be {@code null} or {@link
160     * ConfigurationValue#NULL NULL}.</p>
161     *
162     * @see #value()
163     */
164    private final String name;
165
166    /**
167     * The {@link String} representation of the default value to use
168     * for the configuration value in question if no suitable
169     * configuration value could be found.
170     *
171     * <p>This field will never be {@code null} but may be {@link
172     * ConfigurationValue#NULL NULL}.</p>
173     *
174     * @see #defaultValue()
175     *
176     * @see ConfigurationValue#NULL
177     */
178    private final String defaultValue;
179
180
181    /*
182     * Constructors.
183     */
184
185
186    /**
187     * Creates a new {@link ConfigurationValue.Literal}.
188     *
189     * @param name the name of the configuration value; may be {@code
190     * null} or {@link ConfigurationValue#NULL NULL} in which case
191     * {@code ""} will be used instead
192     *
193     * @param defaultValue the {@link String} representation of the
194     * default value to use when a suitable configuration value cannot
195     * be found; may be {@code null} in which case {@link
196     * ConfigurationValue#NULL NULL} will be used instead
197     *
198     * @see #value()
199     *
200     * @see #defaultValue()
201     */
202    private Literal(final String name, final String defaultValue) {
203      super();
204      this.name = name == null ? "" : NULL.equals(name) ? "" : name;
205      this.defaultValue = defaultValue == null ? NULL : defaultValue;
206    }
207
208
209    /*
210     * Instance methods.
211     */
212
213    
214    /**
215     * Returns the {@link String} representation of the default value to use
216     * for the configuration value in question if no suitable
217     * configuration value could be found.
218     *
219     * <p>This method will never return {@code null} but may return
220     * {@link ConfigurationValue#NULL NULL}.</p>
221     *
222     * @return the {@link String} representation of the default value to use
223     * for the configuration value in question if no suitable
224     * configuration value could be found; never {@code null}
225     * 
226     * @see ConfigurationValue#defaultValue()
227     *
228     * @see ConfigurationValue#NULL
229     */
230    @Override
231    public final String defaultValue() {
232      return this.defaultValue;
233    }
234
235    /**
236     * Returns the name of the configuration value in question.
237     *
238     * <p>This field will never return {@code null} or {@link
239     * ConfigurationValue#NULL NULL}.</p>
240     *
241     * @return the name of the configuration value in question; never
242     * {@code null}
243     *
244     * @see ConfigurationValue#value()
245     *
246     * @see ConfigurationValue#NULL
247     */
248    @Override
249    public final String value() {
250      return this.name;
251    }
252
253
254    /*
255     * Static methods.
256     */
257
258
259    /**
260     * Returns a new {@link Literal} representing a {@link
261     * ConfigurationValue} instance whose {@link
262     * ConfigurationValue#value()} method will return the supplied
263     * {@code name}.
264     *
265     * <p>This method never returns {@code null}.</p>
266     *
267     * @param name the name of the configuration value in question;
268     * may be {@code null} or {@link ConfigurationValue#NULL NULL} in
269     * which case {@code ""} will be used instead
270     *
271     * @return a non-{@code null} {@link Literal}
272     *
273     * @see ConfigurationValue#value()
274     */
275    public static final Literal of(final String name) {
276      return new Literal(name, NULL);
277    }
278
279    /**
280     * Returns a new {@link Literal} representing a {@link
281     * ConfigurationValue} instance whose {@link
282     * ConfigurationValue#value()} method will return the supplied
283     * {@code name} and whose {@link
284     * ConfigurationValue#defaultValue()} method will return the
285     * supplied {@code defaultValue}.
286     *
287     * <p>This method never returns {@code null}.</p>
288     *
289     * @param name the name of the configuration value in question;
290     * may be {@code null} or {@link ConfigurationValue#NULL NULL} in
291     * which case {@code ""} will be used instead
292     *
293     * @param defaultValue the default value for the configuration
294     * value in question; may be {@code null} in which case {@link
295     * ConfigurationValue#NULL NULL} will be used instead
296     *
297     * @return a non-{@code null} {@link Literal}
298     *
299     * @see ConfigurationValue#value()
300     *
301     * @see ConfigurationValue#defaultValue()
302     */
303    public static final Literal of(final String name, final String defaultValue) {
304      return new Literal(name, defaultValue);
305    }
306    
307  }
308  
309}
